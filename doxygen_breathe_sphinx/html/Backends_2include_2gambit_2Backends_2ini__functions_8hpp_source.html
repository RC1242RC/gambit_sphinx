95  void* pHandle = works ? backendInfo().loaded_C_CXX_Fortran_backends.at(be+ver) : NULL ;
96  // Clear error code by calling dlerror()
97  dlerror();
98  // Attempt to obtain a void pointer (pSym) to one of the library symbols.
99  void_voidFptr pSym;
100  for (auto& name : symbol_names)
101  {
102  pSym.ptr = dlsym(pHandle, name.c_str());
103  if (pSym.ptr != NULL) break;
104  }
105  // If using backwards systems missing dlinfo(), like OSX, determine the path to the library with dladdr()
106  #ifndef HAVE_LINK_H
107  // Don't bother trying if the symbol wasn't found in the library anyway.
108  if (pSym.ptr != NULL)
109  {
110  Dl_info info;
111  int dladdr_result = dladdr(pSym.ptr, &info);
112  // Try overriding the path to the library if dladdr seemed to return OK.
113  if (dladdr_result) backendInfo().attempt_backend_path_override(be, ver, info.dli_fname);
114  }
115  #else
116  // Do something inconsequential with the last two args to skip compiler warnings.
117  (void)be;
118  (void)ver;
119  #endif
120  // Hand over the pointer
121  result = reinterpret_cast<T>(pSym.fptr);
122  }
123  catch (std::exception& e) { ini_catch(e); }
