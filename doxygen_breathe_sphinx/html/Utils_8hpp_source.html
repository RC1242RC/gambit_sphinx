215  inline std::map<const HEPUtils::Jet*,bool> generateBTagsMap(const std::vector<const HEPUtils::Jet*>& jets,
216  double bTagEff, double cMissTagEff, double otherMissTagEff,
217  double pTmin = 0., double absEtaMax = DBL_MAX)
218  {
219  std::map<const HEPUtils::Jet*,bool> bTagsMap;
220  for (const HEPUtils::Jet* j : jets)
221  {
222  bool genBTag = false;
223  if((j->pT() > pTmin) && (j->abseta() < absEtaMax))
224  {
225  if(j->btag())
226  {
227  if(random_bool(bTagEff)) { genBTag = true; }
228  }
229  else if(j->ctag())
230  {
231  if(random_bool(cMissTagEff)) { genBTag = true; }
232  }
233  else
234  {
235  if(random_bool(otherMissTagEff)) { genBTag = true; }
236  }
237  }
238  bTagsMap[j] = genBTag;
239  }
240  return bTagsMap;
241  }

283  inline bool object_in_cone(const HEPUtils::Event& e, const HEPUtils::P4& p4, double ptmin, double rmax, double rmin=0.05) {
284  for (const HEPUtils::Particle* p : e.visible_particles())

286  for (const HEPUtils::Jet* j : e.jets())
